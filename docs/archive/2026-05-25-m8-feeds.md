# M8 — Published Feeds Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Publish each user's posts as RSS / Atom / JSON Feed at canonical per-user, per-tag, and site URLs, with incremental cache regeneration on mutation and optional WebSub push.

**Architecture:** Pure serializers in `common::feed`; cache + event-queue storage in `storage::feed_cache` / `storage::feed_events`; axum handlers for lazy regen + 304s; mutation hooks enqueue events; a tokio_cron_scheduler job drains the queue, regenerates feeds, and pings the WebSub hub. WebSub client trait in `common::websub` with HTTP, noop, and capturing impls.

**Tech Stack:** Rust (axum, leptos), `rss` and `atom_syndication` crates (JSON Feed is hand-rolled), reqwest for WebSub, sqlx for DB, tokio_cron_scheduler for the worker.

**Spec:** `docs/superpowers/specs/2026-05-25-m8-feeds-design.md` (read this first).

pub enum FeedFormat {
    Rss,
    Atom,
    Json,
}

impl FeedFormat {
    pub fn ext(self) -> &'static str {
        match self {
            FeedFormat::Rss => "rss",
            FeedFormat::Atom => "atom",
            FeedFormat::Json => "json",
        }
    }
    pub fn content_type(self) -> &'static str {
        match self {
            FeedFormat::Rss => "application/rss+xml; charset=utf-8",
            FeedFormat::Atom => "application/atom+xml; charset=utf-8",
            FeedFormat::Json => "application/feed+json",
        }
    }
}

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub enum FeedSurface {
    Site,
    SiteTag { tag: String },
    User { username: String },
    UserTag { username: String, tag: String },
}

pub fn canonicalize(surface: &FeedSurface, format: FeedFormat) -> String {
    let ext = format.ext();
    match surface {
        FeedSurface::Site => format!("/feed.{ext}"),
        FeedSurface::SiteTag { tag } => format!("/tags/{tag}/feed.{ext}"),
        FeedSurface::User { username } => format!("/~{username}/feed.{ext}"),
        FeedSurface::UserTag { username, tag } => format!("/~{username}/tags/{tag}/feed.{ext}"),
    }
}

pub fn parse(path: &str) -> Option<(FeedSurface, FeedFormat)> {
    // Strip leading '/'
    let rest = path.strip_prefix('/')?;
    // Find ".<ext>" suffix
    let (head, ext) = rest.rsplit_once('.')?;
    let format = match ext {
        "rss" => FeedFormat::Rss,
        "atom" => FeedFormat::Atom,
        "json" => FeedFormat::Json,
        _ => return None,
    };
    // Head must end in "/feed"
    let surface_part = head.strip_suffix("/feed")?;
    if surface_part.is_empty() {
        return Some((FeedSurface::Site, format));
    }
    // tags/:tag
    if let Some(tag) = surface_part.strip_prefix("tags/") {
        if tag.is_empty() || tag.contains('/') {
            return None;
        }
        return Some((FeedSurface::SiteTag { tag: tag.to_string() }, format));
    }
    // ~:username[/tags/:tag]
    if let Some(after_tilde) = surface_part.strip_prefix('~') {
        if let Some((username, tag_part)) = after_tilde.split_once("/tags/") {
            if username.is_empty() || tag_part.is_empty() || tag_part.contains('/') {
                return None;
            }
            return Some((
                FeedSurface::UserTag {
                    username: username.to_string(),
                    tag: tag_part.to_string(),
                },
                format,
            ));
        }
        if after_tilde.contains('/') || after_tilde.is_empty() {
            return None;
        }
        return Some((FeedSurface::User { username: after_tilde.to_string() }, format));
    }
    None
}

use chrono::{DateTime, Duration, Utc};

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub struct HybridWindow {
    pub min_items: u32,
    pub min_days: u32,
}

impl Default for HybridWindow {
    fn default() -> Self {
        Self { min_items: 20, min_days: 30 }
    }
}

pub trait HasPublishedAt {
    fn published_at(&self) -> DateTime<Utc>;
}

impl HybridWindow {
    pub fn cutoff_date(&self, now: DateTime<Utc>) -> DateTime<Utc> {
        now - Duration::days(self.min_days as i64)
    }

    /// `posts` must be ordered by `published_at DESC`.
    /// Returns the prefix of posts where, for index `i`,
    /// `i < min_items` OR `published_at >= cutoff_date`.
    pub fn select<'a, P: HasPublishedAt>(&self, posts: &'a [P], now: DateTime<Utc>) -> &'a [P] {
        let cutoff = self.cutoff_date(now);
        let mut end = 0usize;
        for (i, p) in posts.iter().enumerate() {
            if (i as u32) < self.min_items || p.published_at() >= cutoff {
                end = i + 1;
            } else {
                break;
            }
        }
        &posts[..end]
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[derive(Debug)]
    struct P(DateTime<Utc>);
    impl HasPublishedAt for P {
        fn published_at(&self) -> DateTime<Utc> {
            self.0
        }
    }

    fn at(days_ago: i64, now: DateTime<Utc>) -> P {
        P(now - Duration::days(days_ago))
    }

    #[test]
    fn empty_input_returns_empty() {
        let w = HybridWindow { min_items: 20, min_days: 30 };
        let now = Utc::now();
        assert!(w.select::<P>(&[], now).is_empty());
    }

    #[test]
    fn fewer_than_min_items_returns_all() {
        let w = HybridWindow { min_items: 20, min_days: 30 };
        let now = Utc::now();
        let posts: Vec<P> = (0..5).map(|i| at(i, now)).collect();
        assert_eq!(w.select(&posts, now).len(), 5);
    }

    #[test]
    fn quiet_blog_includes_min_items_even_if_all_older_than_min_days() {
        let w = HybridWindow { min_items: 20, min_days: 30 };
        let now = Utc::now();
        // 25 posts, all 100+ days ago
        let posts: Vec<P> = (0..25).map(|i| at(100 + i, now)).collect();
        // First 20 included because i < min_items; remaining 5 dropped (both predicates fail)
        assert_eq!(w.select(&posts, now).len(), 20);
    }

    #[test]
    fn busy_blog_includes_full_day_window_beyond_min_items() {
        let w = HybridWindow { min_items: 20, min_days: 30 };
        let now = Utc::now();
        // 50 posts all within the last 30 days
        let posts: Vec<P> = (0..50).map(|i| at(i / 2, now)).collect();
        assert_eq!(w.select(&posts, now).len(), 50);
    }

    #[test]
    fn union_stops_at_first_post_failing_both() {
        let w = HybridWindow { min_items: 3, min_days: 30 };
        let now = Utc::now();
        // posts at days_ago = [1, 2, 3, 100, 200] (5 posts)
        // i=0 (1d ago): i<3 → keep
        // i=1 (2d):     i<3 → keep
        // i=2 (3d):     i<3 → keep

        let posts = vec![at(1, now), at(2, now), at(3, now), at(100, now), at(200, now)];
        let kept = w.select(&posts, now);
        assert_eq!(kept.len(), 3);
    }
}

    pub id: i64, // last_post_id input to ETag
    pub title: Option<String>,
    pub permalink: String,
    pub summary: Option<String>,
    pub content_html: String,
    pub published_at: DateTime<Utc>,
    pub updated_at: DateTime<Utc>,
    pub tags: Vec<String>,
}

impl crate::feed::window::HasPublishedAt for FeedItem {
    fn published_at(&self) -> DateTime<Utc> {
        self.published_at
    }
}

/// Strong validator. Format: `"sha256-<hex32>"`.
pub fn feed_etag(items: &[FeedItem], generated_at: DateTime<Utc>) -> String {
    let mut hasher = Sha256::new();
    let max_updated = items.iter().map(|i| i.updated_at).max().unwrap_or(generated_at);
    let last_id = items.last().map(|i| i.id).unwrap_or(0);
    hasher.update(max_updated.to_rfc3339().as_bytes());
    hasher.update(b"|");
    hasher.update((items.len() as u64).to_le_bytes());
    hasher.update(b"|");
    hasher.update(last_id.to_le_bytes());
    let digest = hasher.finalize();
    let hex: String = digest.iter().take(16).map(|b| format!("{b:02x}")).collect();
    format!("\"sha256-{hex}\"")
}

        .map(|i| {
            ItemBuilder::default()
                .title(i.title.clone())
                .link(Some(i.permalink.clone()))

    builder
        .title(meta.title.clone())
        .link(meta.canonical_url.clone())
        .description(meta.description.clone().unwrap_or_default())
        .last_build_date(Some(meta.updated_at.to_rfc2822()))

pub fn render_atom(meta: &FeedMetadata, items: &[FeedItem]) -> String {
    let mut links = vec![
        LinkBuilder::default()
            .href(meta.canonical_url.clone())
            .rel("alternate".to_string())
            .build(),
        LinkBuilder::default()
            .href(meta.self_url.clone())
            .rel("self".to_string())
            .build(),

            LinkBuilder::default()
                .href(hub.clone())
                .rel("hub".to_string())
                .build(),
        );
    }

    let entries: Vec<_> = items
        .iter()
        .map(|i| {
            let mut entry = EntryBuilder::default()
                .id(i.permalink.clone())
                .title(i.title.clone().unwrap_or_default())
                .updated(i.updated_at.fixed_offset())
                .published(Some(i.published_at.fixed_offset()))
                .links(vec![LinkBuilder::default()
                    .href(i.permalink.clone())
                    .rel("alternate".to_string())
                    .build()])
                .content(Some(
                    ContentBuilder::default()
                        .content_type(Some("html".to_string()))
                        .value(Some(i.content_html.clone()))
                        .build(),
                ))
                .categories(
                    i.tags
                        .iter()
                        .map(|t| CategoryBuilder::default().term(t.clone()).build())
                        .collect::<Vec<_>>(),
                )
                .build();
            if let Some(s) = &i.summary {
                entry.set_summary(Some(s.clone().into()));
            }
            entry
        })
        .collect();

    let feed = FeedBuilder::default()
        .title(meta.title.clone())
        .id(meta.self_url.clone())
        .updated(meta.updated_at.fixed_offset())
        .subtitle(meta.description.clone().map(Into::into))
        .links(links)
        .entries(entries)
        .build();

    feed.to_string()
}

use async_trait::async_trait;
use chrono::{DateTime, Duration, Utc};
use sqlx::SqlitePool;

use crate::feed_events::{
    FeedEventError, FeedEventRecord, FeedEventStatus, FeedEventStorage,
};

pub struct SqliteFeedEventStorage {
    pool: SqlitePool,
}

impl SqliteFeedEventStorage {
    pub fn new(pool: SqlitePool) -> Self {
        Self { pool }
    }
}

fn parse_status(s: &str) -> FeedEventStatus {
    match s {
        "pending" => FeedEventStatus::Pending,
        "claimed" => FeedEventStatus::Claimed,
        "done" => FeedEventStatus::Done,
        _ => FeedEventStatus::Failed,
    }
}

#[async_trait]
impl FeedEventStorage for SqliteFeedEventStorage {
    async fn enqueue(&self, feed_url: &str) -> Result<i64, FeedEventError> {
        let id = sqlx::query!(
            "INSERT INTO feed_events (feed_url) VALUES (?) RETURNING id",
            feed_url
        )
        .fetch_one(&self.pool)
        .await?
        .id;
        Ok(id)
    }

    async fn claim_pending_batch(
        &self,
        limit: usize,
        lease_timeout: Duration,
    ) -> Result<Vec<FeedEventRecord>, FeedEventError> {
        let now = Utc::now();
        let lease_cutoff = now - lease_timeout;
        let limit_i = limit as i64;

        // SQLite does not support UPDATE ... RETURNING with LIMIT in all versions;
        // we use a CTE to pick eligible IDs, then UPDATE by id list.
        let mut tx = self.pool.begin().await?;
        let ids: Vec<i64> = sqlx::query_scalar!(
            r#"
            SELECT id FROM feed_events
            WHERE (status = 'pending' AND next_attempt_at <= ?)
               OR (status = 'claimed' AND claimed_at < ?)
            ORDER BY next_attempt_at ASC
            LIMIT ?
            "#,
            now,
            lease_cutoff,
            limit_i,
        )
        .fetch_all(&mut *tx)
        .await?;

        if ids.is_empty() {
            tx.commit().await?;
            return Ok(vec![]);
        }

        // Build "?, ?, ?" placeholders
        let placeholders = std::iter::repeat("?").take(ids.len()).collect::<Vec<_>>().join(",");
        let sql = format!(
            "UPDATE feed_events SET status = 'claimed', claimed_at = ? WHERE id IN ({placeholders})"
        );
        let mut q = sqlx::query(&sql).bind(now);
        for id in &ids {
            q = q.bind(*id);
        }
        q.execute(&mut *tx).await?;

        let select_sql = format!(
            "SELECT id, feed_url, status, attempts, last_error, next_attempt_at, claimed_at, created_at, regenerated_at, pinged_at \
             FROM feed_events WHERE id IN ({placeholders})"
        );
        let mut sq = sqlx::query(&select_sql);
        for id in &ids {
            sq = sq.bind(*id);
        }
        use sqlx::Row;
        let rows = sq.fetch_all(&mut *tx).await?;
        tx.commit().await?;

        let records = rows
            .into_iter()
            .map(|r| FeedEventRecord {
                id: r.get("id"),
                feed_url: r.get("feed_url"),
                status: parse_status(r.get::<&str, _>("status")),
                attempts: r.get::<i64, _>("attempts") as i32,
                last_error: r.get("last_error"),
                next_attempt_at: r.get("next_attempt_at"),
                claimed_at: r.get("claimed_at"),
                created_at: r.get("created_at"),
                regenerated_at: r.get("regenerated_at"),
                pinged_at: r.get("pinged_at"),
            })
            .collect();
        Ok(records)
    }

    async fn mark_regenerated(&self, ids: &[i64]) -> Result<(), FeedEventError> {
        if ids.is_empty() {
            return Ok(());
        }
        let now = Utc::now();

Postgres supports `UPDATE … RETURNING *` natively, so the claim can be done in a single statement:

```rust
async fn claim_pending_batch(
    &self,
    limit: usize,
    lease_timeout: Duration,
) -> Result<Vec<FeedEventRecord>, FeedEventError> {
    let now = Utc::now();
    let lease_cutoff = now - lease_timeout;
    let limit_i = limit as i64;
    let rows = sqlx::query_as!(
        FeedEventRecord,
        r#"
        WITH eligible AS (
          SELECT id FROM feed_events
          WHERE (status = 'pending' AND next_attempt_at <= $1)
             OR (status = 'claimed' AND claimed_at < $2)
          ORDER BY next_attempt_at ASC
          LIMIT $3
          FOR UPDATE SKIP LOCKED
        )
        UPDATE feed_events SET status = 'claimed', claimed_at = $1
        WHERE id IN (SELECT id FROM eligible)
        RETURNING
          id,
          feed_url,
          status AS "status: FeedEventStatusRaw",
          attempts,
          last_error,
          next_attempt_at,
          claimed_at,
          created_at,
          regenerated_at,
          pinged_at
        "#,
        now, lease_cutoff, limit_i,
    )
    .fetch_all(&self.pool)
    .await?;
    Ok(rows.into_iter().map(Into::into).collect())
}

The method returns posts ordered by `published_at DESC`, scoped by surface (site / site-tag / user / user-tag), filtered to `published_at IS NOT NULL AND published_at <= now`, capped at `min_items` posts but **also** including all posts where `published_at >= cutoff`. Since both predicates apply at the SQL layer:

```sql
-- SQLite, user feed
SELECT p.*
FROM posts p
JOIN users u ON u.id = p.author_id
WHERE u.username = ?
  AND p.published_at IS NOT NULL
  AND p.published_at <= ?
  AND (p.published_at >= ?            -- min_days cutoff
       OR p.id IN (
         SELECT id FROM posts
         WHERE author_id = u.id AND published_at IS NOT NULL AND published_at <= ?
         ORDER BY published_at DESC
         LIMIT ?
       ))
ORDER BY p.published_at DESC;
```

That's one approach. A simpler one — and what this plan recommends — is to do the union in code using `HybridWindow::select`:

```rust
// Fetch max(min_items, ALL posts since cutoff) in one query, then trim.
// "Fetch up to N posts published since cutoff OR top-min_items, whichever is larger"
// is awkward in SQL; instead fetch the larger of:
//   (a) all posts since cutoff
//   (b) the top min_items
// by querying with: ORDER BY published_at DESC LIMIT min_items*5  (heuristic upper bound)
// then applying HybridWindow::select in Rust.
```

Recommended impl: query with `ORDER BY published_at DESC LIMIT (greatest(min_items, K))` where K is something like "expected max items in min_days" — safer: do **two queries** and merge:

```rust
let by_count: Vec<Post> = /* SELECT … ORDER BY published_at DESC LIMIT min_items */;
let by_window: Vec<Post> = /* SELECT … WHERE published_at >= cutoff ORDER BY published_at DESC */;
// Union by id, preserve descending order
```

For simplicity, code the merge in Rust:

```rust
let cutoff = window.cutoff_date(now);
let by_window = self.fetch_published_for_surface_since(surface, cutoff, now).await?;
let by_count = self.fetch_published_for_surface_top(surface, window.min_items as i64, now).await?;
let mut out: Vec<Post> = by_window;
for p in by_count {
    if !out.iter().any(|q| q.id == p.id) {
        out.push(p);
    }
}
out.sort_by(|a, b| b.published_at.cmp(&a.published_at));
Ok(out)
```

    async fn send_publish(&self, hub_url: &str, feed_url: &str) -> Result<(), WebSubError> {

#[async_trait]
impl WebSubClient for HttpWebSubClient {
    async fn send_publish(&self, hub_url: &str, feed_url: &str) -> Result<(), WebSubError> {
        let form = [("hub.mode", "publish"), ("hub.url", feed_url)];
        let res = self.client.post(hub_url).form(&form).send().await
            .map_err(|e| {
                if e.is_timeout() { WebSubError::Timeout(self.timeout) }
                else { WebSubError::Http(e.to_string()) }

        let status = res.status();
        if status.is_success() {
            Ok(())
        } else {
            Err(WebSubError::HubRefused { status: status.as_u16() })
        }
    }

use std::sync::Arc;

use chrono::Utc;
use common::feed::{
    canonicalize, feed_etag, parse, render_atom, render_json, render_rss, FeedFormat,
    FeedItem, FeedMetadata, FeedSurface, HybridWindow,
};
use storage::feed_cache::{FeedCacheRow, FeedCacheStorage};
use storage::AppState;
use thiserror::Error;

#[derive(Debug, Error)]
pub enum RegenerateError {
    #[error("unparseable feed_url: {0}")]
    BadUrl(String),
    #[error("storage error: {0}")]
    Storage(String),
}

pub async fn regenerate_feed(state: &AppState, feed_url: &str) -> Result<FeedCacheRow, RegenerateError> {
    let (surface, format) = parse(feed_url).ok_or_else(|| RegenerateError::BadUrl(feed_url.into()))?;
    let window = HybridWindow {
        min_items: state.site_config.get_feeds_min_items().await.map_err(|e| RegenerateError::Storage(e.to_string()))?,
        min_days: state.site_config.get_feeds_min_days().await.map_err(|e| RegenerateError::Storage(e.to_string()))?,
    };
    let hub_url = state.site_config.get_feeds_websub_hub_url().await.map_err(|e| RegenerateError::Storage(e.to_string()))?;
    let now = Utc::now();
    let posts = state.posts.list_published_in_window(&surface, &window, now).await.map_err(|e| RegenerateError::Storage(e.to_string()))?;

    // Build FeedItem list. Tag fetching: batch per (post_id) — use existing posts.fetch_tags_for if available; else one query per post for now and note as a perf TODO if it shows up in benchmarks.
    let items: Vec<FeedItem> = build_feed_items(state, &posts).await.map_err(|e| RegenerateError::Storage(e.to_string()))?;

    let base = state.site.base_url.trim_end_matches('/').to_string();
    let self_url = format!("{base}{}", percent_encode_path(feed_url));
    let canonical_url = match &surface {
        FeedSurface::Site => format!("{base}/"),
        FeedSurface::SiteTag { tag } => format!("{base}/tags/{}/", urlencoding::encode(tag)),
        FeedSurface::User { username } => format!("{base}/~{username}/"),
        FeedSurface::UserTag { username, tag } => format!("{base}/~{username}/tags/{}/", urlencoding::encode(tag)),
    };

    let updated_at = items.iter().map(|i| i.updated_at).max().unwrap_or(now);
    let title = compute_title(state, &surface).await.map_err(|e| RegenerateError::Storage(e.to_string()))?;

    let meta = FeedMetadata {
        title,
        description: None,
        canonical_url,
        self_url,
        hub_url: hub_url.clone(),
        updated_at,
    };

    let body = match format {
        FeedFormat::Rss => render_rss(&meta, &items),
        FeedFormat::Atom => render_atom(&meta, &items),
        FeedFormat::Json => render_json(&meta, &items),
    };
    let etag = feed_etag(&items, now);

    let row = FeedCacheRow {
        feed_url: feed_url.to_string(),
        body,
        etag,
        content_type: format.content_type().to_string(),
        updated_at,
        generated_at: now,
    };
    state.feed_cache.upsert(row.clone()).await.map_err(|e| RegenerateError::Storage(e.to_string()))?;
    Ok(row)
}

fn percent_encode_path(path: &str) -> String {
    use percent_encoding::{utf8_percent_encode, AsciiSet, CONTROLS};
    const PATH_RESERVED: &AsciiSet = &CONTROLS.add(b' ').add(b'"').add(b'<').add(b'>').add(b'`').add(b'#').add(b'?');
    utf8_percent_encode(path, PATH_RESERVED).to_string()
}

async fn build_feed_items(state: &AppState, posts: &[storage::Post]) -> Result<Vec<FeedItem>, storage::PostsError> {
    let mut items = Vec::with_capacity(posts.len());
    for p in posts {
        let tags = state.posts.list_tags_for_post(p.id).await?;
        items.push(FeedItem {
            id: p.id,
            title: p.title.clone(),
            permalink: post_permalink(state, p),
            summary: None,
            content_html: p.body_html.clone(),
            published_at: p.published_at.expect("filtered to published"),
            updated_at: p.updated_at,
            tags,
        });
    }
    Ok(items)
}

fn post_permalink(state: &AppState, post: &storage::Post) -> String {
    let base = state.site.base_url.trim_end_matches('/');
    format!("{base}/~{}/posts/{}", post.author_username, post.id)
}

async fn compute_title(state: &AppState, surface: &FeedSurface) -> Result<String, storage::SiteConfigError> {
    // Use existing site.title / per-user display name lookups.
    let site_title = state.site.title.clone();
    Ok(match surface {
        FeedSurface::Site => site_title,
        FeedSurface::SiteTag { tag } => format!("{site_title} — #{tag}"),
        FeedSurface::User { username } => format!("{site_title} — @{username}"),
        FeedSurface::UserTag { username, tag } => format!("{site_title} — @{username} #{tag}"),
    })
}

use axum::{
    extract::{Path, State},
    http::{header, HeaderMap, HeaderValue, StatusCode},
    response::{IntoResponse, Response},
};
use common::feed::{canonicalize, FeedFormat, FeedSurface};
use storage::AppState;

use super::regenerate::regenerate_feed;

#[derive(Debug, serde::Deserialize)]
pub struct FormatPath { pub ext: String }

fn parse_format(ext: &str) -> Option<FeedFormat> {
    match ext {
        "rss" => Some(FeedFormat::Rss),
        "atom" => Some(FeedFormat::Atom),
        "json" => Some(FeedFormat::Json),
        _ => None,
    }
}

async fn serve(state: Arc<AppState>, headers: HeaderMap, surface: FeedSurface, format: FeedFormat) -> Response {
    let feed_url = canonicalize(&surface, format);
    let row = match state.feed_cache.get(&feed_url).await {
        Ok(Some(row)) => row,
        Ok(None) => match regenerate_feed(&state, &feed_url).await {
            Ok(row) => row,
            Err(e) => return (StatusCode::INTERNAL_SERVER_ERROR, e.to_string()).into_response(),
        },
        Err(e) => return (StatusCode::INTERNAL_SERVER_ERROR, e.to_string()).into_response(),
    };

    if let Some(etag) = headers.get(header::IF_NONE_MATCH) {
        if etag.to_str().ok() == Some(row.etag.as_str()) {
            return StatusCode::NOT_MODIFIED.into_response();
        }
    }
    if let Some(ims) = headers.get(header::IF_MODIFIED_SINCE) {
        if let Some(t) = ims.to_str().ok().and_then(|s| chrono::DateTime::parse_from_rfc2822(s).ok()) {
            if row.updated_at <= t.with_timezone(&chrono::Utc) {
                return StatusCode::NOT_MODIFIED.into_response();
            }
        }
    }

    let mut resp_headers = HeaderMap::new();
    resp_headers.insert(header::CONTENT_TYPE, HeaderValue::from_str(&row.content_type).unwrap());
    resp_headers.insert(header::ETAG, HeaderValue::from_str(&row.etag).unwrap());
    resp_headers.insert(header::LAST_MODIFIED, HeaderValue::from_str(&row.updated_at.to_rfc2822()).unwrap());
    resp_headers.insert(header::CACHE_CONTROL, HeaderValue::from_static("public, max-age=300"));
    (StatusCode::OK, resp_headers, row.body).into_response()
}

  `BackupConfig` pattern; single source of truth for what "site
  settings" means; lets `SiteIdentity` grow new fields without
  threading more parameters through callers.
- **What might join later:** `site.description`, default theme,
  registration policy display copy, footer text. Each can be added as
  a `SiteIdentity` field with a getter that supplies a default — the
  struct is the contract, the key constants are an implementation
  detail.
- **`feeds.*` keys deliberately NOT folded in here:** they're a
  separate concept (per-feed rendering) and are already accessed as a
  cohesive group inside `regenerate_feed`. A future `FeedConfig`
  struct could mirror this pattern; tracked separately if it becomes
  worthwhile.
- **Validation on `base_url`:** intentionally light (non-empty +
  trim + strip trailing slash). Operators can paste a malformed URL
  and the feed XML will simply be wrong; we don't go further than the
  existing `feeds.websub_hub_url` validation, which is the same.

use common::feed::{canonicalize, FeedFormat, FeedSurface};
use storage::{AppState, Post};

pub async fn enqueue_feed_events(
    state: &AppState,
    post: &Post,
    old_tags: &[String],
) -> Result<(), storage::feed_events::FeedEventError> {
    let mut surfaces: Vec<FeedSurface> = vec![
        FeedSurface::Site,
        FeedSurface::User { username: post.author_username.clone() },
    ];
    let mut all_tags: BTreeSet<&str> = BTreeSet::new();
    for t in &post.tags { all_tags.insert(t); }
    for t in old_tags { all_tags.insert(t); }
    for tag in all_tags {
        surfaces.push(FeedSurface::SiteTag { tag: tag.to_string() });
        surfaces.push(FeedSurface::UserTag {
            username: post.author_username.clone(),
            tag: tag.to_string(),
        });
    }

    for surface in &surfaces {
        for format in [FeedFormat::Rss, FeedFormat::Atom, FeedFormat::Json] {
            let url = canonicalize(surface, format);
            state.feed_events.enqueue(&url).await?;
        }
    }
    Ok(())
}

`server/src/feed/worker.rs`:

```rust
use std::collections::HashMap;
use std::sync::Arc;
use std::time::Duration;

use chrono::Utc;
use storage::AppState;
use storage::feed_events::{FeedEventError, FeedEventRecord};

use super::regenerate::regenerate_feed;

const BATCH_LIMIT: usize = 200;
const LEASE_TIMEOUT: Duration = Duration::from_secs(300);
const PING_TIMEOUT: Duration = Duration::from_secs(5);
const BACKOFFS_SECS: &[u64] = &[60, 300, 1800, 7200, 7200, 7200];

pub async fn tick(state: Arc<AppState>) {
    let claimed = match state.feed_events.claim_pending_batch(BATCH_LIMIT, chrono::Duration::from_std(LEASE_TIMEOUT).unwrap()).await {
        Ok(c) => c,
        Err(e) => {
            tracing::error!(error = %e, "feed worker claim failed");
            return;
        }
    };
    if claimed.is_empty() { return; }

    // Group by feed_url.
    let mut groups: HashMap<String, Vec<FeedEventRecord>> = HashMap::new();
    for rec in claimed {
        groups.entry(rec.feed_url.clone()).or_default().push(rec);
    }

    let hub_url = state.site_config.get_feeds_websub_hub_url().await.ok().flatten();

    for (feed_url, recs) in groups {
        let ids: Vec<i64> = recs.iter().map(|r| r.id).collect();
        let started = std::time::Instant::now();
        match regenerate_feed(&state, &feed_url).await {
            Ok(row) => {
                if let Err(e) = state.feed_events.mark_regenerated(&ids).await {
                    tracing::error!(error = %e, feed_url, "mark_regenerated failed");
                    continue;
                }
                tracing::info!(
                    feed_url,
                    item_bytes = row.body.len(),
                    duration_ms = started.elapsed().as_millis(),
                    "feed.regen.completed"
                );

                let attempt = recs.iter().map(|r| r.attempts).max().unwrap_or(0) + 1;
                match &hub_url {
                    Some(hub) => {
                        let absolute = format!("{}{}", state.site.base_url.trim_end_matches('/'), feed_url);
                        tracing::info!(feed_url, hub, attempt, "feed.websub.ping.attempted");
                        let result = state.websub.send_publish(hub, &absolute).await;
                        match result {
                            Ok(()) => {
                                tracing::info!(feed_url, hub, attempt, "feed.websub.ping.succeeded");
                                let _ = state.feed_events.mark_pinged(&ids).await;
                            }
                            Err(e) => {
                                let next_attempt_idx = attempt as usize - 1;
                                if next_attempt_idx >= BACKOFFS_SECS.len() {
                                    tracing::warn!(feed_url, hub, "feed.websub.ping.exhausted");
                                    let _ = state.feed_events.mark_exhausted(&ids, &e.to_string()).await;
                                } else {
                                    let delay = chrono::Duration::seconds(BACKOFFS_SECS[next_attempt_idx] as i64);
                                    let next = Utc::now() + delay;
                                    tracing::warn!(feed_url, hub, attempt, error = %e, "feed.websub.ping.failed");
                                    let _ = state.feed_events.mark_failed(&ids, &e.to_string(), next).await;
                                }
                            }
                        }
                    }
                    None => {
                        // No hub configured — treat as complete.
                        let _ = state.feed_events.mark_pinged(&ids).await;
                    }
                }
            }
            Err(e) => {
                tracing::error!(error = %e, feed_url, "feed.regen.failed");
                let attempt = recs.iter().map(|r| r.attempts).max().unwrap_or(0) + 1;
                let next_attempt_idx = attempt as usize - 1;
                if next_attempt_idx >= BACKOFFS_SECS.len() {
                    let _ = state.feed_events.mark_exhausted(&ids, &e.to_string()).await;
                } else {
                    let next = Utc::now() + chrono::Duration::seconds(BACKOFFS_SECS[next_attempt_idx] as i64);
                    let _ = state.feed_events.mark_failed(&ids, &e.to_string(), next).await;
                }
            }
        }
    }
}

pub fn register(scheduler: &mut tokio_cron_scheduler::JobScheduler, state: Arc<AppState>) -> Result<(), tokio_cron_scheduler::JobSchedulerError> {
    let job = tokio_cron_scheduler::Job::new_repeated_async(Duration::from_secs(10), move |_uuid, _l| {
        let state = state.clone();
        Box::pin(async move { tick(state).await; })
    })?;
    futures::executor::block_on(scheduler.add(job))?;
    Ok(())
}
```

use async_trait::async_trait;
use chrono::Utc;
use std::path::PathBuf;

use super::{WebSubClient, WebSubError};

pub struct FileCapturingWebSubClient { path: PathBuf }

impl FileCapturingWebSubClient {
    pub fn new(path: PathBuf) -> Self { Self { path } }
}

#[async_trait]
impl WebSubClient for FileCapturingWebSubClient {
    async fn send_publish(&self, hub_url: &str, feed_url: &str) -> Result<(), WebSubError> {
        let line = format!("{}\t{}\t{}\n", Utc::now().to_rfc3339(), hub_url, feed_url);

        Ok(())
    }
}

const PATH = process.env.JAUNDER_WEBSUB_CAPTURE_FILE!;

export async function readWebSubPings(): Promise<{ ts: string; hub: string; feed: string }[]> {
  try {
    const text = await readFile(PATH, "utf8");
    return text.split("\n").filter(Boolean).map(line => {
      const [ts, hub, feed] = line.split("\t");
      return { ts, hub, feed };
    });
  } catch (e: any) {
    if (e.code === "ENOENT") return [];
    throw e;
  }
}

export async function waitForNewPing(previousCount: number, timeoutMs = 30_000): Promise<{ ts: string; hub: string; feed: string }> {
  const start = Date.now();
  while (Date.now() - start < timeoutMs) {
    const pings = await readWebSubPings();
    if (pings.length > previousCount) return pings[pings.length - 1];
    await new Promise(r => setTimeout(r, 200));
  }
  throw new Error(`waitForNewPing: no ping after ${timeoutMs}ms (previousCount=${previousCount})`);
